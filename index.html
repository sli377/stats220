My meme

I am taking stats paper(Stats220) so I think it would be fun to create a meme about stats, and when create this meme I just remembered how my mate and I wanted to use the knowledge of statistic to win the lotto. That is why I created this meme.

library("magick")

# read images
imyes <- image_read("image/yes.jpeg") %>%
  image_scale(500)

imno <- image_read("image/no.jpg") %>%
  image_scale(500)

# create text
yes_text <- image_blank(width = 500,
                          height = 300,
                          color = "#FFFFFF") %>%
  image_annotate(text = "Learn stats to win lotto",
                 color = "#000000",
                 size = 50,
                 font = "Impact",
                 gravity = "center")

no_text <- image_blank(width = 500,
                          height = 300,
                          color = "#FFFFFF") %>%
  image_annotate(text = "Learn stats for science",
                 color = "#000000",
                 size = 50,
                 font = "Impact",
                 gravity = "center")

# combine image and text
first_row <- c(imno, no_text) %>%
  image_append()

second_row <- c(imyes, yes_text) %>%
  image_append()

meme <- c(first_row, second_row) %>%
  image_append(stack = TRUE)

# save meme as png
image_write(meme,"my_meme.png")

My animated GIF

The reason I create this animate is I often feel less confident, so this animate is for myself.

library("magick")

circle_image <- image_read("image/heart.jpg") %>%
  image_scale(500)

# creating each frame, where the text comes from the counts vector
frame1 <- circle_image %>%
  image_annotate(text = "You", 
                 color = "#DBE9FA",
                 size = 200, 
                 font = "Impact",
                 gravity = "center") %>%
  image_scale(200) %>%
  image_extent("500x500")

frame2 <- circle_image %>%
  image_annotate(text = "are", 
                 color = "#DBE9FA", 
                 size = 200, 
                 font = "Impact",
                 gravity = "center") %>%
  image_scale(300) %>%
  image_extent("500x500")

frame3 <- circle_image %>%
  image_annotate(text = "the", 
                 color = "#DBE9FA", 
                 size = 200, 
                 font = "Impact",
                 gravity = "center") %>%
  image_scale(400) %>%
  image_extent("500x500")

frame4 <- circle_image %>%
  image_annotate(text = "best", 
                 color = "#DBE9FA", 
                 size = 200, 
                 font = "Impact",
                 gravity = "center") %>%
  image_scale(500)

# putting the frames in order using a vector
frames <- c(frame1, frame2, frame3, frame4)

# creating an animation
animation <- image_animate(frames, fps = 1)

# save animation as gif
image_write(animation,"my_animate.gif")
